模块管理

仅workplus管理员可用
  1. 添加模块
  2. 删除模块
  3. 更新模块
  4. 查询模块
  5. 查询模块列表

添加模块

url: /block

method: POST

body:

{
    "type": "banner", //模块5种类型  banner、 pcTabList、 pcAppMenu、 list、 sudoku
    "pcTabList": [],
    "pcAppMenu": [],
    "list": { //普通列表
        "url": "",
        "list": [],
        "style": ""
    },
    "sudoku": {//九宫格
        "list": [],
        "style": ""
    },
    "banner": { //banner横幅
        "list": []
    },
    "toolBtns": {//标题工具按钮
        "text": [
            {
                "url": "",
                "text": "",
                "mediaId": ""
            }
        ],
        "icon": [
            {
                "url": "",
                "text": "",
                "mediaId": ""
            },
            {
                "url": "",
                "text": "",
                "mediaId": ""
            }
        ],
        "type": "icon"//类型  icon或者text
    },
    "mediaId": "", //模块图标
    "name": ""//模块名称
}

result:

{
    "status": "ok",
    "msg": "operation success",
    "results": {
        "__v": 0,
        "_id": "5b1a22e76d93ae4d8ede4953",
        ...results
    }
}
